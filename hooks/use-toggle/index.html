# useToggle

A hook that switch value between two states.

The API is different from ahooks.

# Examples

# Basic usage

false

Basic usageclick the button, look at the value.
复制成功!
<template>
  <div>
    <p>{{ state }}</p>
    <button @click="toggle()">点击</button>
  </div>
</template>

<script lang="ts">
import { useToggle } from 'ahooks-vue';

export default {
  setup() {
    const { state, toggle } = useToggle();

    return {
      state,
      toggle,
    };
  },
};
</script>

# API

const [state, { toggle, setLeft, setRight }] = useToggle(
  defaultValue?: boolean,
);

const [state, { toggle, setLeft, setRight }] = useToggle(
  defaultValue: any = false,
  reverseValue?: any,
);
