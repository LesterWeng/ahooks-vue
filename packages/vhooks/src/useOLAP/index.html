# 基本用法

基本用法分析多维数据
复制成功!
<template>
  <div></div>
</template>

<script lang="ts">
import { useOLAP } from 'ahooks-vue';
import { data } from './data';

export default {
  setup() {
    const { cube, cubeSettings } = useOLAP(data);

    return {
      cube,
      cubeSettings,
    };
  },
};
</script>
