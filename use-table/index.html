# useTable

Hooks that handle Table-related logic, like frontend pagination, sort and search。

New hook.

# defaultParams

const defaultParams = {
  page: {
    index: 1,
    size: 10,
  },
  search: {
    text: '',
    isReg: false,
    keys: [] as string[],
  },
  sort: {
    key: '',
    direction: 'ascend',
    compareFn: defaultCompareFn,
  },
};

function defaultCompareFn(a: any, b: any) {
  return a < b ? -1 : 1;
}
