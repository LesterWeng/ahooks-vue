    # useFullscreen

    一个用于处理 dom 全屏的 Hook

    # 代码演示

    # 基本用法

    isFullscreen: false
    复制成功!
    <template>
  <div ref="elementRef" style="padding: 16px; background: #fff">
    <div
      style="
        width: 300px;
        height: 180px;
        background: rgba(0, 1, 0, 0.1);
        margin-bottom: 10px;
      "
    />
    <button @click="toggleFull()">点击</button>
    <span> isFullscreen: {{ isFullscreen }}</span>
  </div>
</template>

<script lang="ts">
import { useFullscreen } from '@dewfall/vhooks';
import { ref } from 'vue';

export default {
  setup() {
    const elementRef = ref(null);
    const { isFullscreen, toggleFull } = useFullscreen(elementRef);

    return {
      isFullscreen,
      toggleFull,
      elementRef,
    };
  },
};
</script>

    isFullscreen: false
    复制成功!
    <template>
  <div id="container" style="padding: 16px; background: #fff">
    <div
      style="
        width: 300px;
        height: 180px;
        background: rgba(0, 1, 0, 0.1);
        margin-bottom: 10px;
      "
    />
    <button @click="setFull()">全屏</button>
    <button @click="exitFull()">缩小</button>
    <button @click="toggleFull()">切换</button>
    <span> isFullscreen: {{ isFullscreen }}</span>
  </div>
</template>

<script lang="ts">
import { useFullscreen } from '@dewfall/vhooks';
export default {
  setup() {
    const { isFullscreen, toggleFull, setFull, exitFull } = useFullscreen(() =>
      document.getElementById('container'),
    );

    return {
      isFullscreen,
      toggleFull,
      setFull,
      exitFull,
    };
  },
};
</script>

    # API

    const [isFullscreen, { setFull, exitFull, toggleFull }] = useFullScreen(target, options?:Options);

    # Params

    参数说明类型默认值
    targetDOM 节点或者 Ref 对象HTMLElement | () => HTMLElement | React.MutableRefObject-
    options设置(可选)Options-

    # Options

    参数说明类型默认值
    onExitFull监听退出全屏()=>void-
    onFull监听全屏()=>void-

    # Result

    参数说明类型
    isFullscreen是否全屏boolean
    setFull设置全屏()=>void
    exitFull退出全屏()=>void
    toggleFull切换全屏()=>void
