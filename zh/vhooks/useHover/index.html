    # useFullscreen

    一个用于追踪 dom 元素是否有鼠标悬停的 Hook

    # 基本用法

    isHovering: false
    基本用法使用 ref 设置需要需要监听的元素。
    复制成功!
    <template>
  <div>
    <div
      ref="divRef"
      style="width: 200px; height: 200px; background: rgba(0, 0, 0, 0.2)"
    ></div>

    isHovering: <span style="color: blue">{{ isHovering }}</span>
  </div>
</template>

<script lang="ts">
import { useHover } from '@dewfall/vhooks';
import { ref } from 'vue';

export default {
  setup() {
    const divRef = ref(null);
    const isHovering = useHover(divRef);

    return {
      isHovering,
      divRef,
    };
  },
};
</script>

    isHovering: false
    传入Dom元素传入 function 并返回一个 dom 元素。
    复制成功!
    <template>
  <div>
    <div
      id="divRef"
      style="width: 200px; height: 200px; background: rgba(0, 0, 0, 0.2)"
    ></div>

    isHovering: <span style="color: blue">{{ isHovering }}</span>
  </div>
</template>

<script lang="ts">
import { useHover } from '@dewfall/vhooks';

export default {
  setup() {
    const isHovering = useHover(() => document.getElementById('divRef'));

    return {
      isHovering,
    };
  },
};
</script>

    # API

    const isHovering = useHover(target, {
  onEnter,
  onLeave,
});

    # 结果

    参数说明类型
    isHovering判断鼠标元素是否处于 hover 元素boolean

    # 参数

    参数说明类型默认值
    targetDOM 节点或者 Ref 对象(() => HTMLElement) | HTMLElement | React.RefObject-
    onEnter监听进入 hover()=>void-
    onLeave监听离开 hover()=>void-
