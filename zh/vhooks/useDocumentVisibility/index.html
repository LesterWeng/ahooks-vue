    # useDocumentVisibility

    可以获取页面可见状态的 Hook。

    visibilityState API

    # 代码演示

    # 基本用法

    isVisible:
    基本用法监听 document 的可见状态
    复制成功!
    <template>
  <div>isVisible: {{ isVisible }}</div>
</template>

<script lang="ts">
import { useDocumentVisibility } from '@dewfall/vhooks';
import { watch } from 'vue';

export default {
  setup() {
    const { isVisible, visibilityState } = useDocumentVisibility();

    watch(visibilityState, () => {
      console.log('current visible state =  ' + visibilityState.value);
    });

    return {
      visibilityState,
      isVisible,
    };
  },
};
</script>
